UVa 1025

【分析】 时间是单向流逝的,是一个天然的序,影响到决策的只有当前时间和所处的车站,所以可以用d(i,j)表示时刻i,你在车站j(编号为1-n)，最少还需要等待多长时间,边界条件是d(T,n)=0,其他d(T,i)(i不等于n)为正无穷。有如下三种决策. 决策 1: 等1分钟 决策 2:搭乘往右开的车(如果有)。 决策 3:搭乘往左开的车(如果有)。 【Code】 P268 A Spy in the Metro

#include

#include

#include

#include
using namespace std;
const int inf=0x3f3f3f3f;
int N,T,LN,RN,kase=0;
bool has_train[205][55][2];
int dp[205][55],ti[55];

void solve(){

for(int i=1;i<N;++i) dp\[T\]\[i\]=inf;
dp\[T\]\[N\]=0;

for(int i=T-1;i>=0;i--)
    for(int j=1;j<=N;++j){
        dp\[i\]\[j\]=dp\[i+1\]\[j\]+1;//等待一个单位
        if(j<N&&has_train\[i\]\[j\]\[0\]&&i+ti\[j\]<=T)
            dp\[i\]\[j\]=min(dp\[i\]\[j\],dp\[i+ti\[j\]\]\[j+1\]);//右
        if(j>1&&has_train\[i\]\[j\]\[1\]&&i+ti\[j-1\]<=T)
            dp\[i\]\[j\]=min(dp\[i\]\[j\],dp\[i+ti\[j-1\]\]\[j-1\]);//左
    }

    cout<<"Case Number "<<++kase<<": ";
    if(dp\[0\]\[1\]>=inf) cout<<"impossible\\n";
    else cout<<dp\[0\]\[1\]<<"\\n";

}

int main(){
ios::sync_with_stdio(false);
while(cin>>N&&N){
cin>>T;

    memset(has\_train,0,sizeof(has\_train));

    for(int i=1;i<=N-1;++i)
        cin>>ti\[i\];
    cin>>LN;//左端点向右开的车
    for(int i=1;i<=LN;++i){
        int start;
        cin>>start;
        for(int j=1;j<=N-1;++j){
            if(start<=T)has_train\[start\]\[j\]\[0\]=true;//第t秒第j个站点有车
            start+=ti\[j\];
        }
    }

    cin>>RN;//右端点向左开的车
    for(int i=1;i<=RN;++i){
        int start;
        cin>>start;
        for(int j=N-1;j>=1;--j){
               if(start<=T) has_train\[start\]\[j+1\]\[1\]=true;//第t秒第j个站点有车总共N个站台,往回走
                start+=ti\[j\];
        }
    }
    solve();

}

return 0;

}
