Let's Gank Java - 6st

We can use the typeof string to instance a object. Demo

import java.util.*;
public class GeneraicMethod {
    public static void main(String[] args){
        Date date=BeanUtil.getInstance("java.util.Date");
        System.out.println(date);
    }

}

class BeanUtil{//泛型方法
    //<T> T 反馈类型//参数用到的T都需要放在函数前面
    public static <T> T getInstance( String clzName ){
        try{
            Class c = Class.forName(clzName);
            return (T) c.newInstance();
        }
        catch(ClassNotFoundException ex){}
        catch (InstantiationException ex){}
        catch (IllegalAccessException ex){}
        return null;
    }
}

import java.util.*;
public class GenericTreeClass {
    public static void main(String[] args){
        TNode<String> t = new TNode<>("Roo");
        t.add("Left"); t.add("Middle");
        t.getChild(0).add("aaa");
        t.getChild(0).add("bbb");
        t.traverse();
    }
}

class TNode<T>{
    private T value;
    private ArrayList<TNode<T>> children = new ArrayList<>();

    TNode(T v){ this.value= v; }//构造函数
    public T getValue(){
        return this.value;
    }
    public void add(T v){
        TNode<T> child=new TNode<>(v);
        this.children.add(child);
    }
    public TNode<T> getChild(int i){
        if((i<0) || (i>this.children.size())) return null;
        return (TNode<T>)this.children.get(i);
    }
    public void traverse() {
        //查找所有
        System.out.println(this.value);
        for (TNode child : this.children)
            child.traverse();
    }
}
