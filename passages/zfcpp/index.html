using namespace std;

struct Tree{//以0-26作为Tree的头结点
int num;
Tree* Next[26];
Tree(){
num=0;
int i;
for(i=0;i<26;i++)
Next[i]=NULL;
}
}T;

void In(char str){
Tree H=&T;
for(int i=0;str[i];i++){
if(H->Next[str[i]-‘a’]==NULL)//只有微为空才允许创建
H->Next[str[i]-‘a’]=new Tree;//想调用构造函数必须new一个Tree
H=H->Next[str[i]-‘a’];
H->num++;
}
}

int Se(char *str){
Tree *H=&T;
for(int i=0;str[i];i++){
if(H->Next[str[i]-‘a’]==NULL) return 0;
H=H->Next[str[i]-‘a’];
}
return H->num;
}

int main(){
int N;
char str[11]=””;
scanf(“%d”,&N);

while(N--){
    scanf("%s",str);
    In(str);
}

scanf("%d",&N);
while(N--){
    scanf("%s",str);
    printf("%d\\n",Se(str));
}

return 0;

}**

void kmpN(char x,int len) {

while(j!=-1&&x[i]!=x[j])j=nex[j];
nex[++i]=++j;

int kmpC(char x[],int m,char y[],int n) {
int i=0,j=0,ans=0;
