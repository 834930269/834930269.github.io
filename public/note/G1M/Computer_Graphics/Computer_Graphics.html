# 计算机图形学

# 简易图形成形原理

并非图像直接渲染像素,而是通过计算将图形显示在屏幕上 是通过一个个小三角形实现的 比如我们想要画一个圆形,那么我们只需要通过确定好圆心后计算圆周上的点,然后画许多个小三角形即可,三角形的顶点为: 圆心,相邻的两个圆周上的点

# 代码

用的是EGE框架,只用了最简单的库,没有用它内部的画圆

#include<graphics.h>
#include<stdc++.h>
#define M_PI 3.14159265358979323846   // pi
using namespace std;
const double Deg2Rad = 0.0174532924;

class Enviourment {
public:
	int dimensions;
	Enviourment():dimensions(2) {}
	void initializeEnv(int width,int height) {
		initgraph(width, height);
	}

	void DestroyEnv() {
		closegraph();
	}
};
//Point类
class Vector2 {
public:
	double x, y;

	Vector2(double px, double py):x(px),y(py) {}
	Vector2() {}
	bool operator<(const Vector2 &p) {
		if (x < p.x)return true;
		else if (y < p.y) return true;
		else return false;
	}
	Vector2 operator+(const Vector2 &p) {
		Vector2 v2{0,0};
		v2.x = x + p.x;
		v2.y = y + p.y;
		return v2;
	}
	//坐标意义下的减法,向量意义下的减法需要直接drawline
	Vector2 operator-(const Vector2 &p) {
		Vector2 v2{0,0};
		v2.x = x - p.x;
		v2.y = y - p.y;
		return v2;
	}
	
};

class Line {
public:
	Vector2 start, end;
	Line(Vector2 s, Vector2 e) :start(s), end(e) {}
	//使用坐标draw
	static void DrawALine(Vector2 s2, Vector2 e2) {
		line_f(s2.x, s2.y, e2.x, e2.y);
	}
	//使用暂存坐标draw,用法如缓存路径后draw
	void Draw() {
		line_f(start.x, start.y, end.x, end.y);
	}
};
//设置全局颜色
static class Color {
	//内置有Color enum
public:
	//普通,无Alpha通道
	static void SetColorByRGB(int R,int G,int B) {
		setcolor(EGERGB(R,G,B));
	}
	//添加Alpha通道,即透明度
	static void SetColorByRGBA(int R, int G, int B, int A) {
		setcolor(EGERGBA(R, G, B, A));
	}
	//通过内置的颜色列表设置颜色
	static void SetColorByCOLORSEnum(COLORS c) {
		setcolor(c);
	}
};

//三角形顶点集
class vertices {
public:
	Vector2 vts{0.0,0.0};
	vertices(double x1, double y1) {
		vts.x = x1;
		vts.y = y1;
	}
	vertices(){}
	void SetPosition(double x1, double y1) {
		vts.x = x1;
		vts.y = y1;
	}
};

//面片集,三个顶点为一个面片,这里直接记录顶点下标
class triangles {
public:
	int x_index, y_index, z_index;
	triangles(int xi,int yi,int zi):x_index(xi),y_index(yi),z_index(zi){}
};
//形状,必须实现draw方法,必包含顶点集和面片集合
class shape {
public:
	vector<vertices> Vertices;
	vector<triangles> Triangles;
	virtual void draw(){}
};

class Circle :public shape {
public:
	//细节参数,规定圆周有多少个外顶点,如果少了就不是圆了,是正菱形,默认40
	int Detial;
	double r;
	Vector2 circle_center;
	Circle(double rr, Vector2 cc,int det) :r(rr), circle_center(cc),Detial(det) {
		//创建顶点,圆心在顶点一
		//push圆心
		Vertices.push_back(vertices{ cc.x,cc.y });
		//push外顶点
		//计算pre_radius的弧度值
		double pre_rad = Deg2Rad * 360 / det;
		for (int i = 0; i < det; ++i) {
			double deg = -i * pre_rad;
			double x = cos(deg);
			double y = sin(deg);
			vertices v2;
			v2.SetPosition(x*rr+cc.x, y*rr+cc.y);
			Vertices.push_back(v2);
		}
		
		//添加面片
		for (int i = 1; i < det; ++i) {
			triangles nt{ 0,i,i + 1 };
			Triangles.push_back(nt);
		}
		triangles nt{ 0,det,1 };
		Triangles.push_back(nt);
		//初始化完成
	}
	void draw() {
		//每个面片画三条线
		for (int i = 0; i < Triangles.size(); ++i) {
			Vector2 x= Vertices[Triangles[i].x_index].vts, y= Vertices[Triangles[i].y_index].vts, z= Vertices[Triangles[i].z_index].vts;
			Line l1{ x,y }, l2{ x,z }, l3{y,z};
			l1.Draw();
			l2.Draw();
			l3.Draw();
		}
	}
	void darwNoLineToCenter() {
		//每个面片画一条线,不连接圆心
		for (int i = 0; i < Triangles.size(); ++i) {
			Vector2 x = Vertices[Triangles[i].x_index].vts, y = Vertices[Triangles[i].y_index].vts, z = Vertices[Triangles[i].z_index].vts;
			Line l1{ x,y }, l2{ x,z }, l3{ y,z };
			//l1.Draw();
			//l2.Draw();
			l3.Draw();
		}
	}
};


int main() {
	Enviourment e;
	e.initializeEnv(640,480);
	Circle c{ 200.0,Vector2(300.0,250.0),40};
	Color color;
	color.SetColorByCOLORSEnum(GREEN);
	c.draw();
	//c.darwNoLineToCenter();
	getch();
	return 0;
}

vector<Vector2> ver;
vector<Line> line_list[2];
vector<Vector2> v;
void draw_line(int lim,int index) {
	if (lim == 0) return;
	else {
		cleardevice();
		v.clear();
	}
	for (int i = 0; i < line_list[index].size(); ++i) {
		line_list[index][i].Draw();
	}

	for (int i = 0; i < line_list[index].size(); ++i) {
		v.push_back(line_list[index][i].GetMiddle());
	}

	for (int i = 0; i < v.size() - 1; ++i) {
		Line l{ v[i],v[i+1]};
		line_list[(index+1)%2].push_back(l);
	}
	line_list[(index + 1) % 2].push_back(Line{ v[0],v[v.size() - 1] });
	//getch();
	line_list[index].clear();
	draw_line(lim - 1,(index+1)%2);
}

int main() {
	Enviourment e;
	e.initializeEnv(1000, 1000);

	std::random_device rd;  //Will be used to obtain a seed for the random number engine
	std::mt19937 gen(rd()); //Standard mersenne_twister_engine seeded with rd()
	std::uniform_real_distribution<> dis(1.0, 700.0);
	for (int n = 0; n < 500; ++n) {
		double x = dis(gen), y = dis(gen);
		Vector2 v2{x,y};
		ver.push_back(v2);
	}
	for (int i = 0; i < ver.size() - 1; ++i) {
		Line l{ ver[i],ver[i + 1] };
		line_list[0].push_back(l);
	}
	line_list[0].push_back(Line{ ver[0],ver[ver.size() - 1] });
	draw_line(2000,0);
	

	getch();
	return 0;
}
