# 4Sum

# 4SUm

  1. 四数之和
题目

给定一个包含 n 个整数的数组 nums 和一个目标值 target，判断 nums 中是否存在四个元素 a，b，c 和 d ，使得 a + b + c + d 的值与 target 相等？找出所有满足条件且不重复的四元组。

# 解题思路

3Sum基础上再多一层

# AC代码

class Solution {
public:

    vector<vector<int>> twoSum(vector<int>& nums,int target,int start){
        int i=start,j=nums.size()-1;
        vector<vector<int>> temp;
        while(i<j){
            //if(nums[i]>target || nums[j]<target) break;
            int reg=nums[i]+nums[j];
            if(reg==target){
                temp.push_back({nums[i],nums[j]});
                i++;j--;
                while(i<j && nums[i]==nums[i-1]) i++;
                while(i<j && nums[j]==nums[j+1]) j--;
            }else if(reg > target){
                j--;
                while(i<j && nums[j]==nums[j+1]) j--;
            }else{
                i++;
                while(i<j && nums[i]==nums[i-1]) i++;
            }
        }
        return temp;
    }

    vector<vector<int>> threeSum(vector<int>& nums,int target,int start){
        vector<vector<int>> temp;
        for(int i=start;i<nums.size()-2;){
            vector<vector<int>> tag = twoSum(nums,target-nums[i],i+1);
            for(vector<int> vec : tag){
                vec.push_back(nums[i]);
                temp.push_back(vec);
            }
            ++i;
            while(i<nums.size()-2 && nums[i]==nums[i-1]) i++;
        }
        return temp;
    }

    vector<vector<int>> fourSum(vector<int>& nums, int target) {
        vector<vector<int>> temp;
        if(nums.size()<4) return temp;
        sort(nums.begin(),nums.end());
        for(int i=0;i<nums.size()-3;){
            vector<vector<int>> tag = threeSum(nums,target-nums[i],i+1);
            for(vector<int> vec : tag){
                vec.push_back(nums[i]);
                temp.push_back(vec);
            }
            ++i;
            while(i<nums.size()-3 && nums[i]==nums[i-1]) i++;
        }
        return temp;
    }
};
