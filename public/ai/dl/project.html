# 4. 手写数字识别

# 数据集介绍

       mnist数据集,使用tf.keras自带的mnist数据集,使用tf.data对其进行打包

# 代码

import tensorflow as tf
# 设置gpu使用的增长上限
gpus = tf.config.experimental.list_physical_devices(device_type='GPU')
for gpu in gpus：
	tf.config.experimental.set_memory_growth(gpu,True) 
import matplotlib.pyplot as plt
# 这句只有在jupyter中才能使用,为的是直接调用plt.plot就
# 可以直接显示在控制台中,而不需要plt.show()
%matplotlib inline

# 读取数据集
(train_images,train_labels),(test_images,test_labels) = tf.keras.dataset.mnist.load_data()

print((train_images.shape,train_labels.shape))

# 展示第一张图片
plt.imshow(train_images[0])

# 归一化,因为都是0-255
train_images = train_images/255
test_images = test_images/255

# 使用tf.data.dataset进行封装处理
ds_train_img = tf.data.Dataset.from_tensor_slices(train_images)
ds_train_lab = tf.data.Dataset.from_tensor_slices(train_labels)

# 打包
ds_train = tf.data.Dataset.zip((ds_train_img,ds_train_lab))

# 设置一轮训练数据量
batch_size = 64

# 打乱训练数据
ds_train = ds_train.shuffle(10000).repeat().batch(batch_size)

# 设置测试数据
ds_test = tf.data.Dataset.from_tensor_slices((test_images,test_labels))
ds_test.batch(batch_size)

# 建立模型(一个隐含层,输出层用softmax,因为是多分类)

model = tf.keras.Sequential([
	tf.keras.layers.Flatten(input_shape = (28,28)),
	tf.keras.layers.Dense(128, activation = 'relu'),
	tf.keras.layers.Dense(10, activation = 'softmax')
])

# 编译
model.compile(
	optimizer = 'adam',
	loss = 'sparse_categorical_crossentropy' # 非one-hot的softmax用这个
	metrics = ['accuracy'] # 也可以只写acc
)

# 设置每轮多少步,必须整除
# 同batch_size差不多,只不过这个会自动根据数据量和训练轮数的关系
# 来判断每次需要训练多少数据
# 至于数据Dataset这个模块会自动处理的.比如多repeat几次之类的
step_per_epoch = train_images.shape[0]//batch_size

# 训练,加入测试集
model.fit(ds_train,epochs = 5,steps_per_epoch = steps_per_epoch,validation_data = ds_test,validation_steps = 10000//64)

       对于Numpy数据，fit方法的主要参数有：

  • epoch 全面数据跑几次
  • batch_size: 模型每次运行一小部分数据，这一小部分的数量。
  • validation_data:作为验证集的数据，注意这部分数据不可以是测试集数据

       对于DataSet类型的数据，fit方法的主要参数有：

  • steps_per_epoch：每轮epoch训练中的训练步数。这是一个重要的参数。因为Dataset会产出批量的数据（batch of data),所以不需要batch_size参数
  • 当然，同时可以设置验证集数据和验证集步数。

# 代码

import tensorflow as tf
#引入其他包
from tensorflow import keras
import matplotlib.pyplot as plt
%matplotlib inline
import numpy as np
fashion_mnist = keras.datasets.fashion_mnist

(train_images,train_labels),(test_images,test_labels) = fashion_mnist.load_data()
print((train_images.shape,train_labels.shape))

print((test_images.shape,test_labels.shape))
# 扩张数据集的维度,加上一维表示是RGB还是黑白,扩张一维,第四维为一则是黑白
train_images = np.expand_dims(train_images,-1)
print(train_images.shape)
test_images = np.expand_dims(test_images ,-1)

# 建立模型
model = tf.keras.Sequential()
#Conv2D默认填充方式是vaild,尽量不填充
model.add(tf.keras.layers.Conv2D(32,(3,3),input_shape = train_images.shape[1:],activation = 'relu'))
model.add(tf.keras.layers.MaxPool2D())
model.add(tf.keras.layers.Conv2D(64,(3,3),activation='relu'))
model.add(tf.keras.layers.GlobalAveragePooling2D())
model.add(tf.keras.layers.Dense(10,activation = 'softmax'))

model.summary()
# 为什么是26呢?因为CNN和NN是不同的
'''
CNN的过滤层为32个,每个过滤层的大小为3*3
而且没有设置padding,则需要第一层卷积层接受到的shape是由前一层输入和卷积核大小共同决定的

即 28-3+1=26、

MaxPool2D 默认是2*2大小的,池化层
'''

model.compile(
    optimizer = 'adam',
    loss = 'sparse_categorical_crossentropy', #softmax损失函数
    metrics=['acc']
)

history = model.fit(train_images,train_labels,epochs = 30,validation_data = (test_images,test_labels))

history.history.keys()

plt.plot(history.epoch,history.history.get('acc'),label='acc')
plt.plot(history.epoch,history.history.get('val_acc'),label='val_acc')
#  Test数据和train数据的差距有点大,所以可以判断产生了过拟合
