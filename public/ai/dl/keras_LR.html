import pandas as pd
data = pd.read_csv('In.csv')
print(data)

import matplotlib.pyplot as plt
%matplotlib inline
# 在散点图中显示教育与收入的关联
plt.scatter(data.Education,data.Income)

# tip: 必须转成numpy,否则tf不接受
x=data.Education.to_numpy()
y=data.Income.to_numpy()
