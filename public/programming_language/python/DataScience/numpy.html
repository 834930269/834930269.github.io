# 数组创建

例子:

# 数组创建
# 直接创建
a=np.array([2,3,4])
print(str(a))
#类型推断
b=np.array([1,2,3,5.1])
print(str(b) + ' ' + str(b.dtype))

# 序列转换成相应的n维数组,下列两种方式等价
# 即tuple和array均可
b = np.array([(1.5,2,3), (4,5,6)])
print(b)
b = np.array([[1.5,2,3], [4,5,6]])
print(b)

# 可以指定类型
c = np.array([[1,2],[3,4]],dtype=complex)
print(c)

# 一个前提知识: 数组的增长花费很大
# 但很多任务一开始是知道数组的大小的
# 所以可以通过预生成数组来完成相应的操作

# 在zero中传入shape,全部元素为0
print(np.zeros((3,4)))

# 全部元素为1,指定类型
print(np.ones( (2,3,4), dtype=np.int16 ))

# 随机元素
print(np.empty( (2,3) ) )

# arange函数 (最小值,最大值,迭代间距 - 步长Step)
print(np.arange(10,30,5))
print(np.arange(0,2,0.3))

# 但是由于浮点数精度的问题,所以建议使用linspace中接受元素的数量,而非步长
# np.linspace(最小值,最大值,元素数量)
from numpy import pi
print(np.linspace(0,2,9))
x = np.linspace(0,2*pi,100)
f=np.sin(x)
print(f)

# 反三角函数

       arcsin，arccos，和 arctan 函数返回给定角度的 sin，cos 和 tan 的反三角函数。

       这些函数的结果可以通过 numpy.degrees() 函数将弧度转换为角度。

例子


import numpy as np
 
a = np.array([0,30,45,60,90])  
print ('含有正弦值的数组：')
sin = np.sin(a*np.pi/180)  
print (sin)
print ('\n')
print ('计算角度的反正弦，返回值以弧度为单位：')
inv = np.arcsin(sin)  
print (inv)
print ('\n')
print ('通过转化为角度制来检查结果：')
print (np.degrees(inv))
print ('\n')
print ('arccos 和 arctan 函数行为类似：')
cos = np.cos(a*np.pi/180)  
print (cos)
print ('\n')
print ('反余弦：')
inv = np.arccos(cos)  
print (inv)
print ('\n')
print ('角度制单位：')
print (np.degrees(inv))
print ('\n')
print ('tan 函数：')
tan = np.tan(a*np.pi/180)  
print (tan)
print ('\n')
print ('反正切：')
inv = np.arctan(tan)  
print (inv)
print ('\n')
print ('角度制单位：')
print (np.degrees(inv))

# 舍入函数

numpy.around() 函数返回指定数字的四舍五入值。 numpy.around(a,decimals)

参数说明：

a: 数组
decimals: 舍入的小数位数。 默认值为0。 如果为负，整数将四舍五入到小数点左侧的位置


import numpy as np
 
a = np.array([1.0,5.55,  123,  0.567,  25.532])  
print  ('原数组：')
print (a)
print ('\n')
print ('舍入后：')
print (np.around(a))
print (np.around(a, decimals =  1))
print (np.around(a, decimals =  -1))

# 省略表达

b[i] 方括号中的表达式 i 被视为后面紧跟着 : 的多个实例，用于表示剩余轴。NumPy也允许你使用三个点写为b[i,...]。

三个点（ ... ）表示产生完整索引元组所需的冒号。例如，如果 x 是rank为的5数组（即，它具有5个轴），则：

  • x[1,2,...] 相当于 x[1,2,:,:,:]，
  • x[...,3] 等效于 x[:,:,:,:,3]
  • x[4,...,5,:] 等效于 x[4,:,:,5,:]。
>>> c = np.array( [[[  0,  1,  2],               # a 3D array (two stacked 2D arrays)
...                 [ 10, 12, 13]],
...                [[100,101,102],
...                 [110,112,113]]])
>>> c.shape
(2, 2, 3)
>>> c[1,...]                                   # same as c[1,:,:] or c[1]
array([[100, 101, 102],
       [110, 112, 113]])
>>> c[...,2]                                   # same as c[:,:,2]

# np.meshgrid(x,y)

x - x轴取值范围 y - y轴取值范围

两者的笛卡尔积获取范围内所有坐标

# np.frompyfunc(function,count(**args),count(**return))

这个参数是返回一个函数对象,指定 函数名,参数个数,返回值个数

我们函数中定义每一个元素的运算方式,但传递的参数是一个数组
