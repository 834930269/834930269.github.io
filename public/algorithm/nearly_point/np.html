# 平面最近点对

# 举例 POJ 3714 Raid

题意: 有n个塔,n个特工

问: 哪个特工距离附近的塔最近

# 问题

       这个题目即求每两个点之间的最近距离,最终返回最小的.但因为题目表达的是两个点集间的最短距离,所以可以使用平面最近点对的方法计算,每当计算两个的距离是最短的时候,需要判断一下这两个点是否是同一个点集,不是的话则允许更新.

       平面最近点对: 给n个平面上的点,求出最近的两个点的距离,暴力O(n^2).考虑分治,每次拆分为一半,考虑最小值为1.右侧最小值,2.左侧最小值,3.跨边界最小值.如果跨边界最小值还要暴力计算,那优化毫无意义,所以我们有递归得到了之前的最小值d,所以我们只需要在边界的[mid-d,mid+d]内计算点对的距离即可(斜边长大于任意直角边).

       注意,这里的中线是计算和中间节点的距离.计算前先进行排序,然后向右/左扫描即可确定计算范围.

# 代码

#include<iostream>
#include"stdio.h"
#include<algorithm>
#include<cmath>
using namespace std;
typedef long long LL;
const int inf = 0x3f3f3f3f;
const int maxn = 2e5+10;

struct Node {
	LL x, y;
	int f;
	double dis(const Node &p) {
		return sqrt(1.0*(x - p.x)*(x - p.x) + 1.0*(y - p.y)*(y - p.y));
	}
	bool operator <(const Node &p) const {
		return x < p.x;
	}
}s[maxn];
int n;

double solve(int l,int r) {
	if (l == r)return inf;
	int mid = (l + r) >> 1;
	double a = solve(l, mid);
	double b = solve(mid + 1, r);
	double d = min(a, b);
	for (int i = mid; i >= l; --i) {
		//向右搜索,遇到大于之前最小值d的直接cut
		//这一步是确定跨边界计算时的右边
		if (s[mid].x - s[i].x > d) break;
		//这一步是确定跨边界计算时的左边
		for (int j = mid + 1; j <= r; ++j) {
			if (s[j].x - s[i].x > d) break;
			double tmp = s[j].dis(s[i]);
			//确定两个种类不同才允许计算
			if (s[j].f != s[i].f && tmp < d) d = tmp;
		}
	}
	return d;
}


int main() {
	int t;
	scanf_s("%d",&t);
	while (t--) {
		cin >> n;
		for (int i = 1; i <= n; ++i) {
			scanf_s("%lld%lld", &s[i].x, &s[i].y);
			s[i].f = 0; //0代表电塔
		}
		for (int i = n + 1; i <= 2 * n; ++i) {
			scanf_s("%lld%lld", &s[i].x, &s[i].y);
			s[i].f = 1; //1代表agent
		}
		sort(s + 1, s + 1 + 2 * n);
		printf("%.3lf\n",solve(1,2*n));
	}

	return 0;
}
