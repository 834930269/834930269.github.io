# LintCode 669. Coin Change

最值型动态规划

# 题解

f[amount]=min{f(amount-coins[0])+1,f(amount-coins[1])+1,...} 边界条件:

    1. 除了0外 f[coins[i]]=1
    1. 如果最后结果是inf,则输出-1(无解)

# 代码

class Solution {
public:
    const int inf=0x3ffffff;
    int at;
    int f[1000000];
    void solve(vector<int> coins){
        for(int i=0;i<=at;++i){
            for(int j=0;j<coins.size();++j){
                if(i-coins[j]>=0){
                    f[i]=min(f[i],f[i-coins[j]]+1);
                }
            }
        }
    }
    /**
     * @param coins: a list of integer
     * @param amount: a total amount of money amount
     * @return: the fewest number of coins that you need to make up
     */
    int coinChange(vector<int> &coins, int amount) {
        // write your code here
        for(int i=0;i<1000000;++i){
            f[i]=inf;
        }
        for(int i=0;i<coins.size();++i){
            if(coins[i]!=0)
                f[coins[i]]=1;
        }
        at=amount;
        solve(coins);
        if(amount==0) return 0;
        return f[amount]==inf?-1:f[amount];
    }
};

# LintCode 114: Unique Path

计数型动态规划

# 题解

类似于杨辉三角,思路就是只有两个方向的话直接加起来就是全部的方案了

f[i][j]=f[i-1][j]+f[i][j-1] 初始条件: f[0][j]=f[i][0]=1

# 代码

class Solution {
public:
    int f[110][110];
    int solve(int m,int n){
        for(int i=0;i<m;++i) f[i][0]=1;
        for(int i=0;i<n;++i) f[0][i]=1;
        for(int i=1;i<m;++i){
            for(int j=1;j<n;++j){
                f[i][j]=f[i-1][j]+f[i][j-1];
            }
        }
    }
    
    /**
     * @param m: positive integer (1 <= m <= 100)
     * @param n: positive integer (1 <= n <= 100)
     * @return: An integer
     */
    int uniquePaths(int m, int n) {
        // write your code here
        memset(f,0,sizeof(f));
        solve(m,n);
        return f[m-1][n-1];
    }
};

# LintCode 116: Jump Game

       存在型动态规划

# 题解

这道题本质是一个贪心,但是为了解释存在型动态规划,所以用动态规划来做.

       首先我们考虑最末尾,我们如何才能到达n-1呢?

       这时我们需要看看前面是否有一个石头j可以跳到甚至跳过n-1,如果有,我们就可以判断是否有石头可以跳到j就可以了. j不一定是只有一个的

       状态转移方程: f[j] = OR[0,j)(f[i] AND i+ai >= j)

这个方程OR表示只要有一个即成立,括号内需要满足两个条件,一个是f[i]可以跳到,并且从 i可以跳到 j

# 代码

class Solution {
public:

    bool f[1000000];
    
    /**
     * @param A: A list of integers
     * @return: A boolean
     */
    bool canJump(vector<int> &A) {
        // write your code here
        memset(f,false,sizeof(f));
        f[0]=true;
        for(int i=1;i<A.size();++i){
            for(int j=0;j<i;++j){
                if(f[j] && j+A[j]>=i){
                    f[i]=true;
                    break;
                }
            }
        }
        return f[A.size()-1];
    }
};

# LintCode 115: Unique Paths 2

# 题解

有障碍的话,初始条件是遇到障碍后面就无法走到了 转移方程相同,只不过如果是1则一定无法走到

# 代码

class Solution {
public:
    int f[110][110];
    /**
     * @param obstacleGrid: A list of lists of integers
     * @return: An integer
     */
    int uniquePathsWithObstacles(vector<vector<int>> &obstacleGrid) {
        // write your code here
        int m = obstacleGrid.size();
        int n = obstacleGrid[0].size();
        memset(f,0,sizeof(f));
        f[0][0]=1;
        for(int i=0;i<m;++i){
            if(obstacleGrid[i][0]==1) break;
            f[i][0] = 1;
        }
       for(int i=0;i<n;++i){
            if(obstacleGrid[0][i]==1) break;
            f[0][i] = 1;
        }
        for(int i=1;i<m;++i){
            for(int j=1;j<n;++j){
                if(obstacleGrid[i][j]!=1){
                    if(obstacleGrid[i][j-1]!=1) f[i][j]+=f[i][j-1];
                    if(obstacleGrid[i-1][j]!=1) f[i][j]+=f[i-1][j];
                }
            }
        }
        return f[m-1][n-1];
    }
};

# LintCode 515: Paint House

序列型动态规划

# 题解

首先考虑末位状态,末位状态为最后一个屋子的三种染色方案中哪种花费最少

f[i][j]表示第i个屋子第j种染色的最小花费

  • 转移方程: f[i][j] = costs[i][j] + min(f[i-1][(j+1)%3],f[i-1][(j+2)%3])
  • 边界条件: (j=1~3) f[0][j]=costs[0][j]

# 代码

class Solution {
public:
    int f[300000][3];
    /**
     * @param costs: n x 3 cost matrix
     * @return: An integer, the minimum cost to paint all houses
     */
    int minCost(vector<vector<int>> &costs) {
        // write your code here
        int n = costs.size();
        if(n==0) return 0;
        memset(f,0,sizeof(f));
        
        f[0][0] = costs[0][0];
        f[0][1] = costs[0][1];
        f[0][2] = costs[0][2];
        for(int i=1;i<n;++i){
            for(int j=0;j<3;++j){
                f[i][j] = costs[i][j] + min(f[i-1][(j+1)%3],f[i-1][(j+2)%3]);
            }
        }
        
        return min(f[n-1][0],min(f[n-1][1],f[n-1][2]));
    }
};

# LintCode 512: Decode Ways

划分式动态规划

# 题解

详见动态规划目录下的 - 划分式动态规划

# 代码

class Solution {
public:
    int f[300000][3];
    /**
     * @param s: a string,  encoded message
     * @return: an integer, the number of ways decoding
     */
    int numDecodings(string &s) {
        // write your code here
        memset(f,0,sizeof(f));
        int n = s.length();
        if(n==0 || (n==1 && s=="0")) return 0;
        f[0][0]=1;
        f[0][1]=0;
        for(int i=1;i<n;++i){
            if(s[i]!='0') f[i][0] = f[i-1][0] + f[i-1][1];
            int temp = s[i]-'0'+(s[i-1]-'0') * 10;
            if(temp>=10 && temp<=26){
                f[i][1] = f[i-1][0];
            }else f[i][1] = 0;
        }
        
        return f[n-1][0] + f[n-1][1];
    }
};

class Solution {
public:
    int f[300000];
    /**
     * @param s: a string,  encoded message
     * @return: an integer, the number of ways decoding
     */
    int numDecodings(string &s) {
        // write your code here
        memset(f,0,sizeof(f));
        int n = s.length();
        if(n==0 || (n==1 && s=="0")) return 0;
        f[0] = 1;
        for(int i=1;i<=n;++i){
            int t = s[i-1]-'0';
            if(t!=0) f[i] += f[i-1];
            if(i>=2){
                t = s[i-1]-'0' + (s[i-2]-'0') * 10;
                if(t>=10 && t<=26){
                    f[i] += f[i-2];
                }
            }
        }
        
        return f[n];
    }
};

# LintCode 397 最长连续上升子序列

有点坑的一点是它并不是只有连续上升算,连续下降也算.

# 题解

状态转移方程: f[i] = [(如果大于f[i]) f[j-1] + 1] else [1]

答案是最大的f[i]

# 代码

class Solution {
public:

    int temp1,ans1,temp2,ans2;

    /**
     * @param A: An array of Integer
     * @return: an integer
     */
    int longestIncreasingContinuousSubsequence(vector<int> &A) {
        // write your code here
        if(A.size()==0)return 0;
        ans1=temp1=temp2=ans2=1;
        for(int i=1;i<A.size();++i){
            if(A[i]<A[i-1]){
                temp1+=1;
                ans1=max(ans1,temp1);
            }else temp1=1;
            
            if(A[i]>A[i-1]){
                temp2+=1;
                ans2=max(ans2,temp2);
            }else temp2=1;
        }
        
        return max(ans1,ans2);
    }
};

# LintCode 110 Minimum Path Sum

网格动态规划

# 题解

用一个额外数组记录结果,每次找上左两个最小的加上当前的值.和最短路差不多.

# 代码

class Solution {
public:
    int min_ct[1000][1000];

    /**
     * @param grid: a list of lists of integers
     * @return: An integer, minimizes the sum of all numbers along its path
     */
    int minPathSum(vector<vector<int>> &grid) {
        // write your code here
        int n = grid.size();
        int m = grid[0].size();
        min_ct[0][0] = grid[0][0];
        for(int i=1;i<m;++i){
            min_ct[0][i] = (grid[0][i]+min_ct[0][i-1]);
        }
        for(int i=1;i<n;++i){
            min_ct[i][0] = (grid[i][0]+min_ct[i-1][0]);
        }
        
        for(int i=1;i<n;++i){
            for(int j=1;j<m;++j){
                if(min_ct[i-1][j] > min_ct[i][j-1]){
                    min_ct[i][j] = min_ct[i][j-1] + grid[i][j];
                }else{
                    min_ct[i][j] = min_ct[i-1][j] + grid[i][j];
                }
            }
        }
        return min_ct[n-1][m-1];
    }
};

# 滚动数组的优化

只要f[i][j] 仅依赖本行的f[i][x] 和前一行的 f[i-1][y],那么就可以用滚动数组进行压缩

同理,依赖三个也可以

  1. 大胖子数组:

行少但是列特别多,如果只需要按列计算(不需要依赖其他行列),那么就可以用行作为滚动数组

       代码:

class Solution {
public:
    const int inf =0x3f3f3f3f; 
    /**
     * @param grid: a list of lists of integers
     * @return: An integer, minimizes the sum of all numbers along its path
     */
    int minPathSum(vector<vector<int>> &grid) {
        // write your code here
        int m = grid.size();
        int n = grid[0].size();
        int sz = n > m ? n:m;
        int f[2][sz];
        // old is row i-1
		//now is row
        int old =1,now=0;
        int t1,t2;
        for(int i=0;i<m;++i){
            old = now;
            now = 1 - now;
            for(int j=0;j<n;++j){
                if(i==0 && j==0){
                    f[now][j] = grid[i][j];
                    continue;
                }
                f[now][j] = grid[i][j];
                if(i>0){
                    t1 = f[old][j];
                }else{
                    t1 = inf;
                }
                if(j>0){
                    t2 = f[now][j-1];
                }else{
                    t2=inf;
                }
                if(t1<t2){
                    f[now][j] += t1;
                }else{
                    f[now][j] += t2;
                }
            }
        }
        return f[now][n-1];
    }
};

# LintCode 553. Bomb Enemy

# 坑点和题解

注意,m和n都可能为空 注意,最大数据可能一行有500个

题解: 四个方向分别DP,有E才加,为0则等于上一个,如果为W,则直接置空

# 代码

么得一丁丁点的松耦合性

class Solution {
public:
    int Up[500][500];
    int Down[500][500];
    int Left[500][500];
    int Right[500][500];

    /**
     * @param grid: Given a 2D grid, each cell is either 'W', 'E' or '0'
     * @return: an integer, the maximum enemies you can kill using one bomb
     */
    int maxKilledEnemies(vector<vector<char>> &grid) {
        // write your code here
        int n=grid.size();
        if(n==0) return 0;
        int m = grid[0].size();
        memset(Up,0,sizeof(Up));
        memset(Down,0,sizeof(Down));
        memset(Left,0,sizeof(Left));
        memset(Right,0,sizeof(Right));
        //计算向上
        for(int i=0;i<m;++i){
            for(int j=0;j<n;++j){
                if(j==0){
                    if(grid[j][i]!='W' && grid[j][i]=='E'){
                        Up[j][i]+=1;
                    }
                }else{
                    if(grid[j][i]!='W'){
                        if(grid[j][i]=='E')
                            Up[j][i] = Up[j-1][i]+1;
                        else Up[j][i] = Up[j-1][i];
                    }   
                }
            }
        }
        //计算向左
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(j==0){
                    if(grid[i][j]!='W'&& grid[i][j]=='E'){
                        Left[i][j]+=1;
                    }
                }else{
                    if(grid[i][j]!='W'){
                        if(grid[i][j]=='E')
                            Left[i][j] = Left[i][j-1] + 1;
                        else Left[i][j] = Left[i][j-1];
                    }   
                } 
            }
        }
        //计算向右
        for(int i=n-1;i>=0;--i){
            for(int j=m-1;j>=0;--j){
                if(j==m-1){
                    if(grid[i][j]!='W'&& grid[i][j]=='E')
                        Right[i][j]+=1;
                }else{
                    if(grid[i][j]!='W'){
                        if(grid[i][j]=='E')   
                            Right[i][j] = Right[i][j+1] + 1;
                        else Right[i][j] = Right[i][j+1];
                    }
                }
            }
        }
        //计算向下
        for(int i=m-1;i>=0;--i){
            for(int j=n-1;j>=0;--j){
                if(j==n-1){
                    if(grid[j][i]!='W'&& grid[j][i]=='E')
                        Down[j][i] += 1;
                }else{
                    if(grid[j][i]!='W'){
                        if(grid[j][i]=='E')
                            Down[j][i] = Down[j+1][i] + 1;
                        else Down[j][i] = Down[j+1][i];
                    }
                }
            }
        }
        int max_res =0;
        //计算其中最大的
        for(int i=0;i<n;++i){
            for(int j=0;j<m;++j){
                if(grid[i][j]=='0'){
                    int res=Up[i][j] + Right[i][j] + Down[i][j] + Left[i][j];
                    max_res = max(max_res,res);
                }
            }
        }
        return max_res;
    }
};

# LintCode 664 Counting Bits

# 类型与题解

位操作型,题解在DP章节介绍了

# 代码

class Solution {
public:

    int cnt[1500000];
    /**
     * @param num: a non negative integer number
     * @return: an array represent the number of 1's in their binary
     */
    vector<int> countBits(int num) {
        // write your code here
        memset(cnt,0,sizeof(cnt));
        
        for(int i=1;i<=num;++i){
            cnt[i] = cnt[i>>1] + (i%2);
        }
        vector<int> ve(cnt,cnt+num+1);
        return ve;
    }
};
